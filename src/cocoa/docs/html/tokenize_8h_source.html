8//
9// \file tokenize.h
10//
11// \brief Recognizes tokens in the assembler language
12//
13// Tokens are delimited by whitespace (tabs or spaces) or newlines.
14//
15// The following tokens are recognized specially:
16//
17// COMMENTs, which begin with a semicolon and end at a newline,
18// are recognized as a single token. Consecutive lines containing
19// nothing but comments are recognized as multiple tokens, one for
20// each line.
21//
22// LESS_THAN_OR_EQUAL is recognized in preference to LESS_THAN.
23// That is, LESS_THAN is recognized only if the character immediately
24// following the '<' is not '='.
25//
26// GREATER_THAN_OR_EQUAL is recognized in preference to GREATER_THAN.
27// That is, GREATER_THAN is recognized only if the character
28// immediately following the '>' is not '='.
29//
30// INCREMENT is recognized in preference to PLUS_SIGN. That is,
31// PLUS_SIGN is recognized only if the character immediately
32// following the first '+' is not another '+'. The upshot to this
33// is that the text "x++2" will be recognized as the three tokens
34// TEXT -> INCREMENT -> TEXT which likely will result in a parse
35// error. The text "x+ +2" will be recognized as the four tokens
36// TEXT -> PLUS_SIGN -> PLUS_SIGN -> TEXT and would be parsed
37// as adding +2 to x.
38//
39// DECREMENT is recognized in preference to MINUS_SIGN. That is,
40// MINUS_SIGN is recognized only if the character immediately
41// following the first '-' is not another '-'. The upshot to this
42// is that the text "x--" will be recognized as the three tokens
43// TEXT -> DECREMENT -> TEXT which likely will result in a parse
44// error. The text "x- -2" will be recognized as the four tokens
45// TEXT -> MINUS_SIGN -> MINUS_SIGN -> TEXT and would be parsed
46// as subtracting -2 from x.
47//
48// NONE is recognized only if the input text consists entirely of
49// whitespace or is just a newline.
50//
51// TEXT is recognized only if the input text is not empty (only
52// a newline), not whitespace, and does not match any other token.
53//
