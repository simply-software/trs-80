9//
10// \file source_line.h
11//
12// \brief Parses and holds information about a single line of source code.
13//
14// This class does a pre-parse of the line according to the
15// following GRAMMAR:
16//
17// STATEMENT ::= (blank line or comment)
18// | LABEL -> MNEMONIC -> OPERAND
19// | LABEL -> MNEMONIC
20// | MNEMONIC -> OPERAND
21// | MNEMONIC
22// | LINE_NUMBER -> LABEL -> MNEMONIC -> OPERAND
23// | LINE_NUMBER -> MNEMONIC -> OPERAND
24// | LINE_NUMBER
25// ;
26//
27// All rules end with an optional COMMENT. COMMENTs are
28// recognized by a leading semicolon and end at the end of line.
29//
30// Except for COMMENTs and quoted strings, all other tokens are
31// delimited by whitespace; this means that "1,X" would be recognized
32// as a valid OPERAND, "1, X" would be recognized as two separate
33// tokens "1," and "X", and '"1, X"' -- where there are actual
34// double-quotes in the source -- would be recognized as a single
35// token "1, X" (with the quotes).
36//
37// LINE_NUMBER is recognized if the first token on the line begins
38// at column 0 and contains only the digits [0-9]. Line numbers may
39// begin with leading zeroes and are always parsed as base 10.
40//
41// sourceLine does NOT understand the concept of a LABEL, MNEMONIC,
42// or OPERAND. SourceLine only parses a line into an optional
43// line number followed by zero or more non-comment tokens. Comments
44// are ignored. As such, the grammar recognized by this class is
45// more generally and succintly defined as follows, where square
46// brackets indicate optionality:
47//
48// STATEMENT ::= [LINE_NUMBER] [TOKEN] [TOKEN] [TOKEN] [COMMENT]
49//
50// where ('man grep' as a starting point for help):
51// LINE_NUMBER = regex("[0-9]{1,}")
52// TOKEN = regex("[^;\w]{1,}")
53// COMMENT = regex(";.*")
54//
55// \note This class checks for numeric overflow on line numbers out of
56// practicality since we store line numbers in a C type. But we
57// don't check for numeric overflow on numeric tokens that are
58// not line numbers for three reasons:
59//
60// 1. "numeric" to this class means base-10, whereas cocoa
61// allows users to use binary, octal, and hexadecimal
62// notations, all of which are parsed elsewhere;
63// 2. What it means to "overflow" is not well-defined here,
64// as, for example, an fcb mnemonic (form constant byte)
65// would overflow at 256.
66// 3. If we checked for overflow here, presumably we would
67// also need to check for underflow, which requires parsing
68// negative numbers, but this class does not recognize '-'
69// specially since line numbers cannot be negative.
70//
